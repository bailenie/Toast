### 3.2 签到成功提示自动消失 → AC-205, BR-019

**问题**：`SignCalendar.tsx` 中签到成功提示没有1.5秒自动消失逻辑

**修正方案**：在 `SignCalendar.tsx` 中添加定时器

```typescript
// client/src/components/game/SignCalendar.tsx

// 在 handleSign 函数中添加
const handleSign = async () => {
  if (isSignedToday || loading) return;

  try {
    setLoading(true);
    setMessage('');
    const res = await api.post(`/circles/${circleId}/sign`);
    if (res.data.success) {
      setIsSignedToday(true);
      setCoinBalance(res.data.data.coinBalance);
      setMessage(res.data.data.message);
      
      // 新增：1.5秒后自动消失
      setTimeout(() => {
        setMessage('');
      }, 1500);
      
      // 更新签到日期列表
      const today = new Date().toISOString().split('T')[0];
      setSignInDates(prev => [...prev, today]);
    }
  } catch (err: any) {
    const errorMsg = err.response?.data?.message || '签到失败';
    setMessage(errorMsg);
  } finally {
    setLoading(false);
  }
};
```

// 新增状态
const [ownedDecorations, setOwnedDecorations] = useState<OwnedDecoration[]>([]);

// 在 loadData 中获取已购买装饰
const loadData = useCallback(async () => {
  if (!activeCircleId) return;
  try {
    const [statusRes, cardsRes, leaderboardRes, decorationsRes] = await Promise.all([
      api.get(`/moyu/status?circleId=${activeCircleId}`),
      api.get('/moyu/cards'),
      api.get(`/moyu/leaderboard?circleId=${activeCircleId}`),
      api.get(`/decorations?circleId=${activeCircleId}`),  // 新增
    ]);
    
    // ... 现有代码
    
    // 新增：获取已购买装饰
    if (decorationsRes.data.success) {
      const purchased = decorationsRes.data.data.decorations
        .filter((d: any) => d.isPurchased)
        .map((d: any) => ({ decorationId: d.id, icon: d.icon }));
      setOwnedDecorations(purchased);
    }

    console.error('加载数据失败:', err);
